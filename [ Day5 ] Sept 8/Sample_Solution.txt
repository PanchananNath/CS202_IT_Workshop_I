import java.util.Scanner;

class Faculty {
    String name;

    public Faculty(String name) {
        this.name = name;
    }

    public String toString() {
        return name;
    }
}

class Student {
    String name;
    int rollNo;
    double age;
    Faculty facultyAdvisor;
    String grade;
    String college;

    public Student(Faculty advisor) {
        this.facultyAdvisor = advisor;
    }

    public void updateStudentRecord(Scanner sc) {
        System.out.print("Enter name: ");
        this.name = sc.nextLine();
        System.out.print("Enter roll number: ");
        this.rollNo = Integer.parseInt(sc.nextLine());
        System.out.print("Enter age: ");
        this.age = Double.parseDouble(sc.nextLine());
        System.out.print("Enter grade: ");
        this.grade = sc.nextLine();
        System.out.print("Enter college: ");
        this.college = sc.nextLine();
    }

    public void getStudentDetails() {
        System.out.println("Name: " + name);
        System.out.println("Roll No: " + rollNo);
        System.out.println("Age: " + age);
        System.out.println("Grade: " + grade);
        System.out.println("College: " + college);
        System.out.println("Advisor: " + getAdvisor());
    }

    public Faculty getAdvisor() {
        return facultyAdvisor;
    }
}

public class UniversitySystem {
    static Student[] students = new Student[5];
    static Faculty[] faculties = new Faculty[2];
    static int studentCount = 0;
    static int facultyCount = 0;

    static {
        for (int i = 0; i < students.length; i++) {
            students[i] = null;
        }
        for (int i = 0; i < faculties.length; i++) {
            faculties[i] = null;
        }
    }

    public static void addFaculty(Scanner sc) {
        if (facultyCount >= faculties.length) {
            System.out.println("Faculty limit reached.");
            return;
        }
        System.out.print("Enter faculty name: ");
        String name = sc.nextLine();
        faculties[facultyCount++] = new Faculty(name);
        System.out.println("Faculty added.");
    }

    public static void deleteFaculty(Scanner sc) {
        System.out.print("Enter faculty name to delete: ");
        String name = sc.nextLine();
        for (int i = 0; i < facultyCount; i++) {
            if (faculties[i] != null && faculties[i].name.equals(name)) {
                faculties[i] = null;
                System.out.println("Faculty deleted.");
                return;
            }
        }
        System.out.println("Faculty not found.");
    }

    public static void addStudent(Scanner sc) {
        if (studentCount >= students.length) {
            System.out.println("Student limit reached.");
            return;
        }
        System.out.print("Enter advisor name: ");
        String advisorName = sc.nextLine();
        Faculty advisor = null;
        for (Faculty f : faculties) {
            if (f != null && f.name.equals(advisorName)) {
                advisor = f;
                break;
            }
        }
        if (advisor == null) {
            System.out.println("Advisor not found.");
            return;
        }
        Student s = new Student(advisor);
        s.updateStudentRecord(sc);
        students[studentCount++] = s;
        System.out.println("Student added.");
    }

    public static void deleteStudent(Scanner sc) {
        System.out.print("Enter roll number to delete: ");
        int roll = Integer.parseInt(sc.nextLine());
        for (int i = 0; i < studentCount; i++) {
            if (students[i] != null && students[i].rollNo == roll) {
                students[i] = null;
                System.out.println("Student deleted.");
                return;
            }
        }
        System.out.println("Student not found.");
    }

    public static void updateStudentRecord(Scanner sc) {
        System.out.print("Enter roll number to update: ");
        int roll = Integer.parseInt(sc.nextLine());
        for (Student s : students) {
            if (s != null && s.rollNo == roll) {
                s.updateStudentRecord(sc);
                System.out.println("Record updated.");
                return;
            }
        }
        System.out.println("Student not found.");
    }

    public static void getStudentDetails(Scanner sc) {
        System.out.print("Enter roll number: ");
        int roll = Integer.parseInt(sc.nextLine());
        for (Student s : students) {
            if (s != null && s.rollNo == roll) {
                s.getStudentDetails();
                return;
            }
        }
        System.out.println("Student not found.");
    }

    public static void getStudentMonitor(Scanner sc) {
        System.out.print("Enter roll number: ");
        int roll = Integer.parseInt(sc.nextLine());
        for (Student s : students) {
            if (s != null && s.rollNo == roll) {
                System.out.println("Advisor: " + s.getAdvisor());
                return;
            }
        }
        System.out.println("Student not found.");
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int choice;
        do {
            System.out.println("\n--- MENU ---");
            System.out.println("1) Add Student record");
            System.out.println("2) Delete student");
            System.out.println("3) Add faculty");
            System.out.println("4) Delete Faculty");
            System.out.println("5) Update Student Record");
            System.out.println("6) Get Student Details");
            System.out.println("7) Get Advisor Details");
            System.out.println("8) Exit");
            System.out.print("Enter your choice: ");
            choice = Integer.parseInt(sc.nextLine());

            switch (choice) {
                case 1: addStudent(sc); break;
                case 2: deleteStudent(sc); break;
                case 3: addFaculty(sc); break;
                case 4: deleteFaculty(sc); break;
                case 5: updateStudentRecord(sc); break;
                case 6: getStudentDetails(sc); break;
                case 7: getStudentMonitor(sc); break;
                case 8: System.out.println("Exiting..."); break;
                default: System.out.println("Invalid choice.");
            }
        } while (choice != 8);
        sc.close();
    }
}
