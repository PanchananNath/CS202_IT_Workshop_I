        System.out.println("Enter the elements of the array:");
        for(int i=0; i<n; i++)
        {
            arr[i] = sc.nextInt();
        }

        int[] result= new int[n];
        int[] stack= new int[n];

        int top= -1;

        for(int i=(n-1); i>=0; i--)
        {
            while(top!= -1 && stack[top]<=arr[i])
            top--;

        if(top==-1)
        result[i]=-1;
        else
        result[i]= stack[top];

        stack[++top] = arr[i];
        }
        for (int val : result) 
        {
            System.out.print(val + " ");
        }

    }
}

import java.util.Scanner;

public class ques3 {
    private int[] values;
    private int[] priorities;
    private int size;
    private int capacity;

    public ques3(int capacity) {
        this.capacity = capacity;
        values = new int[capacity];
        priorities = new int[capacity];
        size = 0;
    }

    public void insert(int value) {
        if (size == capacity) {
            System.out.println("Queue is full!");
            return;
        }

        int i;
        for (i = size - 1; i >= 0 && priorities[i] > value; i--) {
            values[i + 1] = values[i];
            priorities[i + 1] = priorities[i];
        }

        values[i + 1] = value;
        priorities[i + 1] = value;
        size++;
    }

    public int extract() {
        if (size == 0) {
            System.out.println("Queue is empty!");
        }
        int highest = values[0];
        for (int i = 1; i < size; i++) {
            values[i - 1] = values[i];
            priorities[i - 1] = priorities[i];
        }
        size--;
        return highest;
    }

    public void display() {
        System.out.print("Priority Queue: { ");
        for (int i = 0; i < size; i++) {
            System.out.print(values[i] + " ");
        }
        System.out.println("}");
    }

    public static void main(String[] args) {
        ques3 pq = new ques3(20);
        Scanner sc = new Scanner(System.in);

        System.out.print("Enter number of elements to insert: ");
        int n = sc.nextInt();

        System.out.println("Enter " + n + " elements:");
        for (int i = 0; i < n; i++) {
            int val = sc.nextInt();
            pq.insert(val);
        }

        System.out.println("Before extract:");
        pq.display();

        int extracted = pq.extract();
        System.out.println("Extracted item with highest priority: " + extracted);

        System.out.println("After extract:");
        pq.display();

        sc.close();
    }
}
